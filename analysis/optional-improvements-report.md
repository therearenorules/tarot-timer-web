**🎨 테마별 색상 정의**:

```typescript

  background: "#1a1625",  // 다크 퍼플 배경
  surface: "#2d1b47",     // 미드 퍼플 표면
  text: "#ffffff"         // 화이트 텍스트
},

라이트모드: {
  primary: "#7b2cbf",     // 동일한 브랜드 컬러

  background: "#ffffff",  // 화이트 배경
  surface: "#f8f9fa",     // 라이트 그레이 표면
  text: "#2d1b47"         // 다크 퍼플 텍스트
}
```
